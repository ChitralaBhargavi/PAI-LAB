# AI-Based Online Exam Proctor Assignment (Simple)

# Input exams
n = int(input("Enter number of exams: "))
exams = []
for _ in range(n):
    eid = input("Exam ID: ")
    time = input("Time slot (e.g., 9-10): ")
    skill = input("Required skill: ")
    exams.append({"id": eid, "time": time, "skill": skill})

# Input proctors
m = int(input("Enter number of proctors: "))
proctors = []
for _ in range(m):
    pid = input("Proctor ID: ")
    av = input("Available slots (space separated): ").split()
    sk = input("Skills (space separated): ").split()
    proctors.append({"id": pid, "available": av, "skills": sk})

# Assignment dictionary
assignment = {}

# Check if a proctor can invigilate the exam
def is_valid(proctor, exam):
    if exam["time"] not in proctor["available"]:
        return False
    if exam["skill"] not in proctor["skills"]:
        return False
    for e, p in assignment.items():
        if p == proctor["id"] and e["time"] == exam["time"]:
            return False
    return True

# Backtracking function
def assign_exam(i=0):
    if i == len(exams):
        return True
    exam = exams[i]
    for p in proctors:
        if is_valid(p, exam):
            assignment[exam["id"]] = p["id"]
            if assign_exam(i + 1):
                return True
            del assignment[exam["id"]]
    return False

# Solve
if assign_exam():
    print("\nAssignment of Exams to Proctors:")
    for e, p in assignment.items():
        print(f"{e} -> {p}")
else:
    print("\nNo valid assignment possible.")
